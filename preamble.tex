    % Espaciado entre párrafos

        \vspace{-2.5em}
        % \textcolor{main}{\rule{\textwidth}{0.4mm}}
        \begin{abstract}
            #2
        \end{abstract}
        \begin{center}
            \textcolor{main}{\rule{\textwidth}{0.4mm}}
        \end{center}
    \fi

    % \HRuleGray

    % Tabla de contenido:
